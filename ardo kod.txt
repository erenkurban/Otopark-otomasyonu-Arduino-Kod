#include <LiquidCrystal.h>
#include <Servo.h>
Servo exitgate;
Servo entergate;
LiquidCrystal lcd(12,11,5,4,3,2);

  int count=4;

void setup() {
  exitgate.attach(6);
  exitgate.write(0);
  Serial.begin(9600);
  pinMode(A0,INPUT);
  pinMode(A1,INPUT);
  lcd.begin(16,2);
  lcd.clear();
  lcd.print("    PARKING ");
  lcd.setCursor(9,1);

  lcd.print("Count: ");
   lcd.setCursor(15,1);
  lcd.print(count);
 
}

void loop() {

  int ff=digitalRead(A0); /*giriþ sensörü */
  int aa=digitalRead(A1); /*çýkýþ sensörü */
  

if(ff==LOW){
  
   delay(1000);
  if(count>0){count--;}
  lcd.setCursor(9,1);
  lcd.print("Count: ");
  
  lcd.setCursor(15,1);
  lcd.print(count);
  delay(2000);
  
  }

  if(aa==LOW){
  exitgate.write(100);
  delay(1000);
  if(count<4){count++;}
  lcd.setCursor(9,1);
  lcd.print("Count: ");
  
  lcd.setCursor(15,1);
  lcd.print(count);
  delay(2000);
  exitgate.write(0);
  }


}
